<!-- /components/sample/sample-card.vue -->
<!-- TEMPLATE — one row in the gallery. Modeled on
     /components/rewards/reward-card.vue: a reactable-card wrapper
     (components/wonderlab/reactable-card.vue) with hover actions, badges,
     and store-backed select/delete. -->
<template>
  <!-- ReactionTargetType has no per-model 'sample' entry — reuse 'resource'
       / 'RESOURCE', or register your real model in stores/reactionStore.ts
       (targetIdKeyMap + the category enum). -->
  <reactable-card
    :selected="activeSelected"
    :compact="compact"
    :show-reaction="showReaction"
    :target-id="sample.id"
    target-type="resource"
    reaction-category="RESOURCE"
    :target-title="sampleTitle"
    @select="selectSample"
  >
    <template #actions>
      <!-- Interact deep-links: the store action selects the row AND hops the
           dashboard to the interact tab. -->
      <button
        v-if="showActions && (activeSelected || compact)"
        class="rounded-full bg-base-100 p-2 text-success shadow transition hover:bg-success hover:text-success-content"
        type="button"
        title="Interact with Sample"
        @click.stop="interactWithSample"
      >
        <Icon name="kind-icon:play" class="h-4 w-4" />
      </button>

      <button
        v-if="showActions && allowEdit && (activeSelected || compact)"
        class="rounded-full bg-base-100 p-2 text-primary shadow transition hover:bg-primary hover:text-primary-content"
        type="button"
        title="Edit Sample"
        @click.stop="emit('edit', sample.id)"
      >
        <Icon name="kind-icon:pencil" class="h-4 w-4" />
      </button>

      <button
        v-if="showActions && allowDelete && (activeSelected || compact)"
        class="rounded-full bg-base-100 p-2 text-error shadow transition hover:bg-error hover:text-error-content"
        type="button"
        title="Delete Sample"
        @click.stop="deleteSample"
      >
        <Icon name="kind-icon:trash" class="h-4 w-4" />
      </button>
    </template>

    <!-- Sample has an optional imageId (ArtImage). This template keeps an
         icon placeholder; for the full base64/thumbnail/path fallback chain,
         copy loadRewardImage from reward-card.vue. -->
    <div
      :class="[
        'relative flex items-center justify-center overflow-hidden rounded-2xl border border-base-300 bg-base-300',
        compact ? 'h-24 w-full' : 'h-36 w-full',
      ]"
    >
      <Icon
        :name="fallbackIcon"
        :class="compact ? 'h-12 w-12' : 'h-20 w-20'"
        class="text-primary/70"
      />

      <div class="absolute left-2 top-2 flex flex-wrap gap-1">
        <span v-if="activeSelected" class="badge badge-primary badge-sm shadow">
          Selected
        </span>
      </div>

      <div
        v-if="activeSelected"
        class="absolute bottom-2 right-2 rounded-full bg-primary p-2 text-primary-content shadow"
      >
        <Icon name="kind-icon:check" class="h-4 w-4" />
      </div>
    </div>

    <div class="flex min-w-0 flex-1 flex-col gap-2">
      <h2
        :class="[
          'font-black leading-tight text-base-content',
          compact ? 'line-clamp-1 text-base' : 'text-lg',
        ]"
        :title="sampleTitle"
      >
        {{ sampleTitle }}
      </h2>

      <p
        v-if="showDescription"
        :class="[
          'text-base-content/70',
          compact ? 'line-clamp-2 text-sm' : 'text-sm',
        ]"
      >
        {{ sample.description || sample.label || 'No description yet.' }}
      </p>

      <div v-if="showMeta" class="flex flex-wrap gap-2">
        <span
          class="badge badge-sm"
          :class="sample.isPublic ? 'badge-info' : 'badge-ghost'"
        >
          {{ sample.isPublic ? 'Public' : 'Private' }}
        </span>

        <span v-if="sample.isMature" class="badge badge-warning badge-sm">
          Mature
        </span>

        <span v-if="sample.type" class="badge badge-outline badge-sm">
          {{ sample.type }}
        </span>
      </div>

      <details
        v-if="showDebug"
        class="rounded-2xl border border-base-300 bg-base-100 p-2"
        @click.stop
      >
        <summary class="cursor-pointer text-xs font-bold text-base-content/70">
          Debug
        </summary>

        <pre class="mt-2 max-h-48 overflow-auto text-xs text-base-content/70">{{
          JSON.stringify(sample, null, 2)
        }}</pre>
      </details>
    </div>
  </reactable-card>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import { useSampleStore, type Sample } from '@/stores/sampleStore'

const props = withDefaults(
  defineProps<{
    sample: Sample
    selected?: boolean
    compact?: boolean
    showActions?: boolean
    showDescription?: boolean
    showMeta?: boolean
    showReaction?: boolean
    showDebug?: boolean
    allowEdit?: boolean
    allowDelete?: boolean
    fallbackIcon?: string
  }>(),
  {
    selected: false,
    compact: false,
    showActions: true,
    showDescription: true,
    showMeta: true,
    showReaction: true,
    showDebug: false,
    allowEdit: true,
    allowDelete: true,
    fallbackIcon: 'kind-icon:box',
  },
)

const emit = defineEmits<{
  select: [id: number]
  edit: [id: number]
  delete: [id: number]
}>()

const sampleStore = useSampleStore()

const activeSelected = computed(() => {
  return props.selected || sampleStore.selectedSample?.id === props.sample.id
})

const sampleTitle = computed(() => {
  return props.sample.title || props.sample.label || `Sample #${props.sample.id}`
})

async function selectSample() {
  await sampleStore.selectSample(props.sample.id)
  emit('select', props.sample.id)
}

async function interactWithSample() {
  await sampleStore.startSampleInteraction(props.sample.id)
}

// Edit is emitted, not handled here — the parent gallery owns
// startEditingSample and the form panel.
async function deleteSample() {
  const result = await sampleStore.deleteSample(props.sample.id)

  if (result.success) {
    emit('delete', props.sample.id)
  }
}
</script>
