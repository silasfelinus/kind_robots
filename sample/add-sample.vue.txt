<!-- /components/sample/add-sample.vue -->
<!-- TEMPLATE — one form for both add and edit, driven by the mode prop and
     bound directly to the store's form ref. Modeled on
     /components/rewards/add-reward.vue. saveSample() routes to create or
     update based on whether the form carries a positive id. -->
<template>
  <div
    class="mx-auto flex w-full max-w-7xl flex-col gap-6 rounded-2xl border border-base-300 bg-base-200 p-4"
  >
    <header class="text-center">
      <h1 class="text-3xl font-bold text-primary md:text-4xl">
        {{ heading }}
      </h1>

      <p class="mt-2 text-sm text-base-content/70">
        {{ subtitle }}
      </p>
    </header>

    <div
      v-if="mode === 'edit' && !sampleStore.selectedSample"
      class="rounded-2xl border border-warning/40 bg-warning/10 p-4 text-warning"
    >
      Select a sample before editing.
    </div>

    <template v-else>
      <!-- Fields mirror the Sample model exactly (see sample/schema.md):
           title, description, label, type, designer, isPublic, isMature.
           userId is stamped at save; imageId is wired through uploadStore —
           copy configureRewardImageUpload from add-reward.vue if your model
           carries art. -->
      <section class="grid grid-cols-1 gap-4 lg:grid-cols-2">
        <label class="form-control">
          <span class="label">
            <span class="label-text font-bold">Title</span>
          </span>

          <input
            v-model="sampleStore.sampleForm.title"
            type="text"
            placeholder="Sample title"
            class="input input-bordered w-full bg-base-100"
          />
        </label>

        <label class="form-control">
          <span class="label">
            <span class="label-text font-bold">Type</span>
          </span>

          <input
            v-model="sampleStore.sampleForm.type"
            type="text"
            placeholder="utility, model, story..."
            class="input input-bordered w-full bg-base-100"
          />
        </label>

        <label class="form-control">
          <span class="label">
            <span class="label-text font-bold">Label</span>
            <span class="label-text-alt text-base-content/50">Optional</span>
          </span>

          <input
            v-model="sampleStore.sampleForm.label"
            type="text"
            placeholder="Short display label"
            class="input input-bordered w-full bg-base-100"
          />
        </label>

        <label class="form-control">
          <span class="label">
            <span class="label-text font-bold">Designer</span>
            <span class="label-text-alt text-base-content/50">Optional</span>
          </span>

          <input
            v-model="sampleStore.sampleForm.designer"
            type="text"
            placeholder="Who made this?"
            class="input input-bordered w-full bg-base-100"
          />
        </label>

        <label class="form-control lg:col-span-2">
          <span class="label">
            <span class="label-text font-bold">Description</span>
            <span class="label-text-alt text-base-content/50">Optional</span>
          </span>

          <textarea
            v-model="sampleStore.sampleForm.description"
            placeholder="Describe what this sample represents..."
            class="textarea textarea-bordered min-h-28 w-full bg-base-100"
          />
        </label>
      </section>

      <!-- Every user-generated content model carries these two flags. -->
      <section class="grid grid-cols-1 gap-4 sm:grid-cols-2">
        <label
          class="label cursor-pointer justify-between rounded-2xl border border-base-300 bg-base-100 px-4 py-3"
        >
          <span class="label-text font-bold">Public</span>

          <input
            v-model="sampleStore.sampleForm.isPublic"
            type="checkbox"
            class="toggle toggle-primary"
          />
        </label>

        <label
          class="label cursor-pointer justify-between rounded-2xl border border-base-300 bg-base-100 px-4 py-3"
        >
          <span class="label-text font-bold">Mature</span>

          <input
            v-model="sampleStore.sampleForm.isMature"
            type="checkbox"
            class="toggle toggle-accent"
          />
        </label>
      </section>

      <!-- choice-manager gives a field reusable community suggestions; the
           matching choiceStore.applyToForm call in saveSample writes the
           picked value into the form. Current forms doing this: add-reward
           ('rewardEssence'), add-scenario ('genre'). Drop this section if
           your model has no choice-driven fields. -->
      <section class="rounded-2xl border border-base-300 bg-base-100 p-4">
        <h2 class="text-xl font-bold text-base-content">Sample Type Choices</h2>

        <p class="mb-4 text-sm text-base-content/70">
          Reusable choice suggestions for the type field.
        </p>

        <choice-manager label="type" model="Sample" />
      </section>

      <div
        v-if="statusMessage"
        class="rounded-2xl border p-3 text-sm"
        :class="
          statusTone === 'error'
            ? 'border-error/40 bg-error/10 text-error'
            : 'border-success/40 bg-success/10 text-success'
        "
      >
        {{ statusMessage }}
      </div>

      <footer
        class="flex flex-col gap-2 sm:flex-row sm:items-center sm:justify-end"
      >
        <!-- hasUnsavedChanges compares the form payload against the selected
             row, so it stays false until the user actually edits something. -->
        <span
          v-if="sampleStore.hasUnsavedChanges"
          class="badge badge-warning badge-outline sm:mr-auto"
        >
          Unsaved changes
        </span>

        <button class="btn btn-ghost rounded-xl" type="button" @click="cancel">
          Cancel
        </button>

        <button
          v-if="mode === 'edit'"
          class="btn btn-ghost rounded-xl"
          type="button"
          @click="resetFromSelected"
        >
          Revert
        </button>

        <button
          v-else
          class="btn btn-ghost rounded-xl"
          type="button"
          @click="resetForAdd"
        >
          Reset
        </button>

        <button
          class="btn btn-success rounded-xl"
          type="button"
          :disabled="sampleStore.isSaving"
          @click="saveSample"
        >
          <span
            v-if="sampleStore.isSaving"
            class="loading loading-spinner loading-sm"
          />
          {{ saveLabel }}
        </button>
      </footer>
    </template>
  </div>
</template>

<script setup lang="ts">
import { computed, onMounted, ref, watch } from 'vue'
import { useChoiceStore } from '@/stores/choiceStore'
import { useSampleStore } from '@/stores/sampleStore'
import { useUserStore } from '@/stores/userStore'

const props = withDefaults(
  defineProps<{
    mode?: 'add' | 'edit'
  }>(),
  {
    mode: 'add',
  },
)

const emit = defineEmits<{
  saved: []
  cancel: []
}>()

const choiceStore = useChoiceStore()
const sampleStore = useSampleStore()
const userStore = useUserStore()

const statusMessage = ref('')
const statusTone = ref<'success' | 'error'>('success')

const mode = computed(() => props.mode)

const heading = computed(() =>
  mode.value === 'edit' ? 'Edit Sample' : 'Create Sample',
)

const subtitle = computed(() =>
  mode.value === 'edit'
    ? 'Update this sample entry.'
    : 'Create a new sample entry.',
)

const saveLabel = computed(() =>
  mode.value === 'edit' ? 'Save Changes' : 'Save Sample',
)

onMounted(async () => {
  await Promise.all([sampleStore.initialize(), choiceStore.initialize()])
  prepareForm()
})

watch(
  () => props.mode,
  () => {
    prepareForm()
  },
)

function prepareForm() {
  if (mode.value === 'edit') {
    resetFromSelected()
    return
  }

  resetForAdd()
}

// startAddingSample deselects and loads the store's default form —
// no hand-built blank object here.
function resetForAdd() {
  sampleStore.startAddingSample()
  statusMessage.value = ''
}

// startEditingSample re-pulls the selected row into the form,
// discarding local edits.
async function resetFromSelected() {
  const id = sampleStore.selectedSample?.id

  if (!id) return

  await sampleStore.startEditingSample(id)
  statusMessage.value = ''
}

function cancel() {
  sampleStore.deselectSample()
  emit('cancel')
}

async function saveSample() {
  if (mode.value === 'edit' && !sampleStore.sampleForm.id) {
    statusTone.value = 'error'
    statusMessage.value = 'No sample selected or invalid ID.'
    return
  }

  if (!sampleStore.sampleForm.title?.trim()) {
    statusTone.value = 'error'
    statusMessage.value = 'Please provide a title.'
    return
  }

  try {
    // Fold any picked choice values into the form before saving.
    choiceStore.applyToForm(sampleStore.sampleForm, 'type', 'Sample')

    // Stamp ownership; guest/system rows use userId 10.
    sampleStore.sampleForm = {
      ...sampleStore.sampleForm,
      userId: sampleStore.sampleForm.userId || userStore.userId || 10,
    }

    const saved = await sampleStore.saveSample()

    if (!saved) {
      throw new Error(sampleStore.error || 'Failed to save sample.')
    }

    statusTone.value = 'success'
    statusMessage.value =
      mode.value === 'edit'
        ? 'Sample updated successfully.'
        : 'Sample saved successfully.'

    emit('saved')
  } catch (error) {
    console.error('Failed to save sample:', error)
    statusTone.value = 'error'
    statusMessage.value =
      error instanceof Error ? error.message : 'Failed to save sample.'
  }
}
</script>
