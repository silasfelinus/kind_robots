<script setup lang="ts">
import { computed, ref } from 'vue'
import { useSampleStore } from '@/stores/sampleStore'
import { useNavStore } from '@/stores/navStore'

type ActionMode = 'preview' | 'launch' | 'inspect' | 'custom'

type ActionModeEntry = {
  value: ActionMode
  title: string
  description: string
  icon: string
}

type StatusTone = 'success' | 'error'

// Must match the manager's registered dashboard key.
const dashboardKey = 'sample' as const

const sampleStore = useSampleStore()
const navStore = useNavStore()

const actionMode = ref<ActionMode>('preview')
const customDirection = ref('')
const statusMessage = ref('')
const statusTone = ref<StatusTone>('success')
const isRunning = ref(false)
const lastOutput = ref('')

// Placeholder actions — swap these for the real verbs of your model.
const actionModes: ActionModeEntry[] = [
  {
    value: 'preview',
    title: 'Preview',
    description: 'Summarize what this sample is ready to do.',
    icon: 'kind-icon:eye',
  },
  {
    value: 'launch',
    title: 'Launch',
    description: 'Prepare this sample for an active workflow.',
    icon: 'kind-icon:rocket',
  },
  {
    value: 'inspect',
    title: 'Inspect',
    description: 'Expose fields, flags, and metadata.',
    icon: 'kind-icon:search',
  },
  {
    value: 'custom',
    title: 'Custom',
    description: 'Use your instruction as the primary action.',
    icon: 'kind-icon:wand',
  },
]

// All reads go through the store — this view owns no model state.
const selectedSample = computed(() => sampleStore.selectedSample)

const selectedSampleTitle = computed(() => {
  return (
    selectedSample.value?.title ||
    selectedSample.value?.label ||
    'No sample selected'
  )
})

const selectedSampleDescription = computed(() => {
  return (
    selectedSample.value?.description ||
    selectedSample.value?.label ||
    'No description provided.'
  )
})

const canRunAction = computed(() => {
  return Boolean(selectedSample.value && !isRunning.value)
})

const normalizedCustomDirection = computed(() => {
  return customDirection.value.trim()
})

const interactionPreview = computed(() => {
  const sample = selectedSample.value

  if (!sample) return ''

  const lines = [
    `Sample: ${selectedSampleTitle.value}`,
    `Type: ${sample.type || 'sample'}`,
    `Label: ${sample.label || 'none'}`,
    `Designer: ${sample.designer || 'none'}`,
    `Description: ${selectedSampleDescription.value}`,
    `Public: ${sample.isPublic ? 'yes' : 'no'}`,
    `Mature: ${sample.isMature ? 'yes' : 'no'}`,
    `Action mode: ${actionMode.value}`,
    normalizedCustomDirection.value
      ? `Custom direction: ${normalizedCustomDirection.value}`
      : '',
  ]

  return lines.filter(Boolean).join('\n')
})

function setStatus(message: string, tone: StatusTone = 'success') {
  statusMessage.value = message
  statusTone.value = tone
}

function backToGallery() {
  navStore.setDashboardTab(dashboardKey, 'gallery')
}

function buildActionOutput() {
  const sample = selectedSample.value

  if (!sample) {
    throw new Error('No sample selected.')
  }

  if (actionMode.value === 'preview') {
    return `Preview ready for ${selectedSampleTitle.value}.`
  }

  if (actionMode.value === 'launch') {
    return `Launch prepared for ${selectedSampleTitle.value}.`
  }

  if (actionMode.value === 'inspect') {
    return [
      `Inspecting ${selectedSampleTitle.value}`,
      `Type: ${sample.type || 'sample'}`,
      `Label: ${sample.label || 'none'}`,
      `Designer: ${sample.designer || 'none'}`,
      `Public: ${sample.isPublic ? 'yes' : 'no'}`,
      `Mature: ${sample.isMature ? 'yes' : 'no'}`,
      `ID: ${sample.id}`,
    ].join('\n')
  }

  if (!normalizedCustomDirection.value) {
    return `Custom action selected for ${selectedSampleTitle.value}, but no custom direction was provided.`
  }

  return [
    `Custom action for ${selectedSampleTitle.value}`,
    normalizedCustomDirection.value,
  ].join('\n')
}

async function runInteraction() {
  if (!canRunAction.value) return

  isRunning.value = true
  statusMessage.value = ''

  try {
    const output = buildActionOutput()

    lastOutput.value = output
    setStatus('Interaction complete.')
  } catch (error) {
    const message =
      error instanceof Error ? error.message : 'Interaction failed.'

    lastOutput.value = ''
    setStatus(message, 'error')
  } finally {
    isRunning.value = false
  }
}

async function copyPreview() {
  if (!interactionPreview.value) return

  await navigator.clipboard.writeText(interactionPreview.value)
  setStatus('Preview copied.')
}

function clearPromptOptions() {
  actionMode.value = 'preview'
  customDirection.value = ''
  statusMessage.value = ''
  lastOutput.value = ''
}
</script>
