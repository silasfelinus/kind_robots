<script setup lang="ts">
// /components/navigation/sample-footer.vue
import { computed, onMounted, ref, watch } from 'vue'
import { useDisplayStore } from '@/stores/displayStore'
import { useSampleModelStore } from '@/stores/sampleModelStore'

const displayStore = useDisplayStore()
const sampleStore = useSampleModelStore()

const footerState = computed(() => displayStore.footerState)
const isCompact = computed(() => footerState.value === 'compact')
const isOpen = computed(() => footerState.value === 'open')

const selectedId = ref(0)

const canSave = computed(() => {
  return Boolean(
    sampleStore.form.title?.trim() && sampleStore.form.type?.trim(),
  )
})

const sampleStatus = computed(() => {
  if (sampleStore.loading) return 'Loading samples...'
  if (sampleStore.isSaving) return 'Saving sample...'
  if (sampleStore.selected?.id) {
    return `Selected sample #${sampleStore.selected.id}`
  }
  return 'Create or select a sample.'
})

function selectSample() {
  if (!selectedId.value) {
    sampleStore.deselectModel()
    return
  }

  sampleStore.selectModel(selectedId.value)
}

function selectSampleById(id: number) {
  selectedId.value = id
  sampleStore.selectModel(id)
}

async function deleteSelectedSample() {
  if (!sampleStore.selected?.id) return

  const id = sampleStore.selected.id
  await sampleStore.deleteModel(id)

  if (selectedId.value === id) {
    selectedId.value = 0
  }
}

watch(
  () => sampleStore.selected?.id,
  (id) => {
    selectedId.value = id ?? 0
  },
)

onMounted(async () => {
  await sampleStore.initialize()
})
</script>
