<template>
  <section class="flex w-full flex-col gap-4 rounded-2xl bg-base-200 p-4">
    <header
      class="rounded-2xl border border-base-300 bg-base-100 p-4 text-center shadow-md"
    >
      <h1 class="text-2xl font-bold text-primary md:text-3xl">
        Sample Interaction
      </h1>

      <p class="mx-auto mt-2 max-w-3xl text-sm text-base-content/70 md:text-base">
        Use the selected sample as a launch point for an action, preview, prompt, or model-specific workflow.
      </p>
    </header>

    <section class="grid grid-cols-1 gap-4 xl:grid-cols-2">
      <article class="rounded-2xl border border-base-300 bg-base-100 p-3">
        <h2 class="mb-2 text-lg font-bold text-base-content">
          Selected Sample
        </h2>

        <div
          v-if="sampleStore.selected"
          class="rounded-2xl border border-primary/30 bg-primary/10 p-3"
        >
          <p class="font-bold text-primary">
            {{ sampleStore.selected.title }}
          </p>

          <p class="mt-2 text-sm text-base-content/70">
            {{ sampleStore.selected.description || sampleStore.selected.label }}
          </p>

          <p class="mt-2 text-xs text-base-content/50">
            Type: {{ sampleStore.selected.type || 'sample' }}
          </p>
        </div>

        <div
          v-else
          class="rounded-2xl border border-base-300 bg-base-200 p-3 text-sm text-base-content/60"
        >
          No sample selected.
        </div>
      </article>

      <article class="rounded-2xl border border-base-300 bg-base-100 p-3">
        <h2 class="mb-2 text-lg font-bold text-base-content">
          Action
        </h2>

        <select v-model="actionMode" class="select select-bordered w-full bg-base-200">
          <option value="preview">Preview</option>
          <option value="launch">Launch</option>
          <option value="inspect">Inspect</option>
          <option value="custom">Custom</option>
        </select>

        <textarea
          v-model="customDirection"
          class="textarea textarea-bordered mt-3 min-h-28 w-full rounded-2xl bg-base-200"
          placeholder="Optional custom instruction..."
        />
      </article>
    </section>

    <section
      v-if="interactionPreview"
      class="rounded-2xl border border-base-300 bg-base-100 p-4 shadow-md"
    >
      <details open>
        <summary class="cursor-pointer text-lg font-bold text-base-content">
          Interaction Preview
        </summary>

        <pre class="mt-3 whitespace-pre-wrap rounded-2xl bg-base-300 p-3 text-sm text-base-content/80">{{ interactionPreview }}</pre>
      </details>
    </section>

    <footer class="flex flex-col gap-2 sm:flex-row sm:justify-end">
      <button
        class="btn btn-ghost rounded-xl"
        type="button"
        @click="clearPromptOptions"
      >
        Reset
      </button>

      <button
        class="btn btn-primary rounded-xl"
        type="button"
        :disabled="!sampleStore.selected"
        @click="copyPreview"
      >
        <Icon name="kind-icon:copy" class="h-5 w-5" />
        Copy
      </button>
    </footer>
  </section>
</template>

<script setup lang="ts">
import { computed, ref } from 'vue'
import { useSampleModelStore } from '@/stores/sampleModelStore'

const sampleStore = useSampleModelStore()

const actionMode = ref<'preview' | 'launch' | 'inspect' | 'custom'>('preview')
const customDirection = ref('')

const interactionPreview = computed(() => {
  const sample = sampleStore.selected

  if (!sample) return ''

  return [
    `Sample: ${sample.title || 'Untitled Sample'}`,
    `Type: ${sample.type || 'sample'}`,
    `Label: ${sample.label || 'none'}`,
    `Component: ${sample.component || 'none'}`,
    `Link: ${sample.link || 'none'}`,
    `Action mode: ${actionMode.value}`,
    customDirection.value.trim()
      ? `Custom direction: ${customDirection.value.trim()}`
      : '',
  ]
    .filter(Boolean)
    .join('\n')
})

async function copyPreview() {
  if (!interactionPreview.value) return

  await navigator.clipboard.writeText(interactionPreview.value)
}

function clearPromptOptions() {
  actionMode.value = 'preview'
  customDirection.value = ''
}
</script>
