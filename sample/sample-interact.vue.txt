    <header
      class="rounded-2xl border border-base-300 bg-base-100 p-4 text-center shadow-md"
    >
      <h1 class="text-2xl font-bold text-primary md:text-3xl">
        Sample Interaction
      </h1>

      <p
        class="mx-auto mt-2 max-w-3xl text-sm text-base-content/70 md:text-base"
      >
        Use the selected sample as a launch point for previews, inspections,
        prompts, or model-specific workflows.
      </p>
    </header>

    <div
      v-if="statusMessage"
      class="rounded-2xl border p-3 text-sm"
      :class="
        statusTone === 'error'
          ? 'border-error/40 bg-error/10 text-error'
          : 'border-success/40 bg-success/10 text-success'
      "
    >
      {{ statusMessage }}
    </div>

    <section
      class="grid min-h-0 flex-1 grid-cols-1 gap-4 overflow-hidden xl:grid-cols-[minmax(0,1fr)_minmax(320px,440px)]"
    >
      <main
        class="flex min-h-0 flex-col gap-4 overflow-hidden rounded-2xl border border-base-300 bg-base-100 p-4 shadow-md"
      >
        <section class="shrink-0">
          <div class="mb-3 flex flex-wrap items-center justify-between gap-2">
            <div>
              <h2 class="text-lg font-bold text-base-content">
                Selected Sample
              </h2>

              <p class="text-sm text-base-content/60">
                This is the current store selection.
              </p>
            </div>

            <div
              v-if="selectedSample"
              class="badge badge-primary badge-outline"
            >
              {{ selectedSampleType }}
            </div>
          </div>

          <article
            v-if="selectedSample"
            class="rounded-2xl border border-primary/30 bg-primary/10 p-4"
          >
            <div class="flex flex-col gap-3 md:flex-row md:items-start">
              <div
                class="flex h-12 w-12 shrink-0 items-center justify-center rounded-2xl border border-primary/30 bg-base-100"
              >
                <Icon :name="selectedSampleIcon" class="h-7 w-7 text-primary" />
              </div>

              <div class="min-w-0 flex-1">
                <h3 class="text-xl font-bold text-primary">
                  {{ selectedSampleTitle }}
                </h3>

                <p class="mt-2 whitespace-pre-wrap text-sm text-base-content/75">
                  {{ selectedSampleDescription }}
                </p>

                <div class="mt-3 grid grid-cols-1 gap-2 text-xs sm:grid-cols-2">
                  <div class="rounded-xl bg-base-100/80 p-2">
                    <span class="font-bold text-base-content/60">Label:</span>
                    {{ selectedSample.label || 'none' }}
                  </div>

                  <div class="rounded-xl bg-base-100/80 p-2">
                    <span class="font-bold text-base-content/60">Component:</span>
                    {{ selectedSample.component || 'none' }}
                  </div>

                  <div class="rounded-xl bg-base-100/80 p-2">
                    <span class="font-bold text-base-content/60">Link:</span>
                    {{ selectedSample.link || 'none' }}
                  </div>

                  <div class="rounded-xl bg-base-100/80 p-2">
                    <span class="font-bold text-base-content/60">ID:</span>
                    {{ selectedSample.id ?? 'none' }}
                  </div>
                </div>
              </div>
            </div>
          </article>

          <div
            v-else
            class="flex min-h-44 flex-col items-center justify-center gap-3 rounded-2xl border border-base-300 bg-base-200 p-4 text-center text-base-content/55"
          >
            <Icon name="kind-icon:sparkles" class="h-14 w-14 text-primary/60" />

            <div>
              <p class="text-lg font-bold">No sample selected</p>
              <p class="mt-1 text-sm">
                Pick a sample from the gallery before launching an interaction.
              </p>
            </div>
          </div>
        </section>

        <section
          class="min-h-0 flex-1 overflow-hidden rounded-2xl border border-base-300 bg-base-200"
        >
          <div
            class="flex items-center justify-between gap-2 border-b border-base-300 bg-base-100 p-3"
          >
            <div>
              <h2 class="text-lg font-bold text-base-content">
                Interaction Preview
              </h2>

              <p class="text-xs text-base-content/50">
                This is the normalized payload generated from the selected
                sample and current action.
              </p>
            </div>

            <button
              class="btn btn-xs btn-ghost rounded-xl"
              type="button"
              :disabled="!interactionPreview"
              @click="copyPreview"
            >
              <Icon name="kind-icon:copy" class="h-4 w-4" />
              Copy
            </button>
          </div>

          <div class="min-h-0 overflow-auto p-3">
            <pre
              v-if="interactionPreview"
              class="min-h-48 whitespace-pre-wrap rounded-2xl bg-base-300 p-3 text-sm leading-relaxed text-base-content/80"
              >{{ interactionPreview }}</pre
            >

            <div
              v-else
              class="flex min-h-48 items-center justify-center rounded-2xl border border-dashed border-base-300 bg-base-100 p-4 text-center text-sm text-base-content/50"
            >
              Select a sample to generate an interaction preview.
            </div>
          </div>
        </section>
      </main>

      <aside class="flex min-h-0 flex-col gap-4 overflow-hidden">
        <section
          class="rounded-2xl border border-base-300 bg-base-100 p-4 shadow-md"
        >
          <h2 class="mb-3 text-lg font-bold text-base-content">
            Action
          </h2>

          <div class="grid grid-cols-1 gap-3">
            <button
              v-for="mode in actionModes"
              :key="mode.value"
              class="btn justify-start rounded-2xl text-left normal-case"
              :class="actionMode === mode.value ? 'btn-primary' : 'btn-outline'"
              type="button"
              :disabled="isRunning"
              @click="actionMode = mode.value"
            >
              <Icon :name="mode.icon" class="h-5 w-5" />

              <span class="flex min-w-0 flex-col items-start">
                <span class="font-bold">{{ mode.title }}</span>
                <span class="text-xs opacity-70">{{ mode.description }}</span>
              </span>
            </button>
          </div>
        </section>

        <section
          class="rounded-2xl border border-base-300 bg-base-100 p-4 shadow-md"
        >
          <h2 class="mb-1 text-lg font-bold text-base-content">
            Custom Direction
          </h2>

          <p class="mb-3 text-sm text-base-content/60">
            Add context, constraints, or the next thing this sample should do.
          </p>

          <textarea
            v-model="customDirection"
            class="textarea textarea-bordered min-h-32 w-full rounded-2xl bg-base-200"
            placeholder="Optional custom instruction..."
            :disabled="isRunning"
          />
        </section>

        <section
          class="rounded-2xl border border-base-300 bg-base-100 p-4 shadow-md"
        >
          <h2 class="mb-3 text-lg font-bold text-base-content">
            Output
          </h2>

          <div
            v-if="lastOutput"
            class="rounded-2xl border border-success/30 bg-success/10 p-3 text-sm text-success"
          >
            <p class="font-bold">Last action</p>
            <p class="mt-1 whitespace-pre-wrap">{{ lastOutput }}</p>
          </div>

          <div
            v-else
            class="rounded-2xl border border-base-300 bg-base-200 p-3 text-sm text-base-content/60"
          >
            No action has been run yet.
          </div>
        </section>

        <footer
          class="mt-auto grid grid-cols-1 gap-2 rounded-2xl border border-base-300 bg-base-100 p-3 shadow-md sm:grid-cols-2"
        >
          <button
            class="btn btn-ghost rounded-xl"
            type="button"
            :disabled="isRunning"
            @click="clearPromptOptions"
          >
            <Icon name="kind-icon:refresh" class="h-5 w-5" />
            Reset
          </button>

          <button
            class="btn btn-primary rounded-xl"
            type="button"
            :disabled="!canRunAction"
            @click="runInteraction"
          >
            <span v-if="isRunning" class="loading loading-spinner loading-sm" />
            <Icon v-else name="kind-icon:play" class="h-5 w-5" />
            Run
          </button>
        </footer>
      </aside>
    </section>
  </section>
</template>

<script setup lang="ts">
import { computed, ref } from 'vue'
import { useSampleModelStore } from '@/stores/sampleModelStore'

type ActionMode = 'preview' | 'launch' | 'inspect' | 'custom'

type ActionModeEntry = {
  value: ActionMode
  title: string
  description: string
  icon: string
}

type StatusTone = 'success' | 'error'

const sampleStore = useSampleModelStore()

const actionMode = ref<ActionMode>('preview')
const customDirection = ref('')
const statusMessage = ref('')
const statusTone = ref<StatusTone>('success')
const isRunning = ref(false)
const lastOutput = ref('')

const actionModes: ActionModeEntry[] = [
  {
    value: 'preview',
    title: 'Preview',
    description: 'Summarize what this sample is ready to do.',
    icon: 'kind-icon:eye',
  },
  {
    value: 'launch',
    title: 'Launch',
    description: 'Prepare this sample for an active workflow.',
    icon: 'kind-icon:rocket',
  },
  {
    value: 'inspect',
    title: 'Inspect',
    description: 'Expose structure, links, labels, and metadata.',
    icon: 'kind-icon:search',
  },
  {
    value: 'custom',
    title: 'Custom',
    description: 'Use your instruction as the primary action.',
    icon: 'kind-icon:wand',
  },
]

const selectedSample = computed(() => sampleStore.selected ?? null)

const selectedSampleTitle = computed(() => {
  return selectedSample.value?.title || selectedSample.value?.label || 'Untitled Sample'
})

const selectedSampleDescription = computed(() => {
  return (
    selectedSample.value?.description ||
    selectedSample.value?.label ||
    'No description provided.'
  )
})

const selectedSampleType = computed(() => {
  return selectedSample.value?.type || 'sample'
})

const selectedSampleIcon = computed(() => {
  return selectedSample.value?.icon || 'kind-icon:sparkles'
})

const canRunAction = computed(() => {
  return Boolean(selectedSample.value && !isRunning.value)
})

const normalizedCustomDirection = computed(() => {
  return customDirection.value.trim()
})

const interactionPreview = computed(() => {
  const sample = selectedSample.value

  if (!sample) return ''

  const lines = [
    `Sample: ${selectedSampleTitle.value}`,
    `Type: ${sample.type || 'sample'}`,
    `Label: ${sample.label || 'none'}`,
    `Description: ${selectedSampleDescription.value}`,
    `Component: ${sample.component || 'none'}`,
    `Link: ${sample.link || 'none'}`,
    `Action mode: ${actionMode.value}`,
    normalizedCustomDirection.value
      ? `Custom direction: ${normalizedCustomDirection.value}`
      : '',
  ]

  return lines.filter(Boolean).join('\n')
})

function setStatus(message: string, tone: StatusTone = 'success') {
  statusMessage.value = message
  statusTone.value = tone
}

function buildActionOutput() {
  const sample = selectedSample.value

  if (!sample) {
    throw new Error('No sample selected.')
  }

  if (actionMode.value === 'preview') {
    return `Preview ready for ${selectedSampleTitle.value}.`
  }

  if (actionMode.value === 'launch') {
    return `Launch prepared for ${selectedSampleTitle.value}.`
  }

  if (actionMode.value === 'inspect') {
    return [
      `Inspecting ${selectedSampleTitle.value}`,
      `Type: ${sample.type || 'sample'}`,
      `Label: ${sample.label || 'none'}`,
      `Component: ${sample.component || 'none'}`,
      `Link: ${sample.link || 'none'}`,
      `ID: ${sample.id ?? 'none'}`,
    ].join('\n')
  }

  if (!normalizedCustomDirection.value) {
    return `Custom action selected for ${selectedSampleTitle.value}, but no custom direction was provided.`
  }

  return [
    `Custom action for ${selectedSampleTitle.value}`,
    normalizedCustomDirection.value,
  ].join('\n')
}

async function runInteraction() {
  if (!canRunAction.value) return

  isRunning.value = true
  statusMessage.value = ''

  try {
    const output = buildActionOutput()

    lastOutput.value = output
    setStatus('Interaction complete.')
  } catch (error) {
    const message =
      error instanceof Error ? error.message : 'Interaction failed.'

    lastOutput.value = ''
    setStatus(message, 'error')
  } finally {
    isRunning.value = false
  }
}

async function copyPreview() {
  if (!interactionPreview.value) return

  await navigator.clipboard.writeText(interactionPreview.value)
  setStatus('Preview copied.')
}

function clearPromptOptions() {
  actionMode.value = 'preview'
  customDirection.value = ''
  statusMessage.value = ''
  lastOutput.value = ''
}
</script>
