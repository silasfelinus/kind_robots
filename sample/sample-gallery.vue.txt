<!-- /components/sample/sample-gallery.vue -->
<!-- TEMPLATE — the browse/select surface. Modeled on
     /components/rewards/reward-gallery.vue (closest current gallery) with the
     mature-content gating every isMature model must carry (see also
     /components/dreams/dream-gallery.vue). Filtering happens here, never in
     the store — the store holds every row it knows about. -->
<template>
  <section
    class="flex h-full min-h-0 w-full flex-col gap-3 rounded-2xl bg-base-300 p-3"
  >
    <header
      v-if="showHeader"
      class="flex shrink-0 flex-col gap-3 rounded-2xl border border-base-300 bg-base-200 p-3"
    >
      <div class="flex items-start justify-between gap-3">
        <div class="min-w-0">
          <h2 class="truncate text-lg font-bold text-base-content">
            {{ title }}
          </h2>

          <p
            v-if="sampleStore.selectedSample"
            class="truncate text-sm text-base-content/70"
          >
            Selected:
            <span class="font-semibold text-primary">
              {{ selectedSampleTitle }}
            </span>
          </p>

          <p v-else class="text-sm text-base-content/60">
            {{ subtitle }}
          </p>
        </div>

        <div class="flex shrink-0 items-center gap-2">
          <span v-if="!isLoading" class="badge badge-ghost">
            {{ filteredSamples.length }}
          </span>

          <button
            v-if="allowAdd"
            class="btn btn-primary btn-sm rounded-xl"
            type="button"
            @click="startAddingSample"
          >
            <Icon name="kind-icon:plus" class="h-4 w-4" />
            <span class="hidden sm:inline">Add</span>
          </button>

          <button
            v-if="allowRefresh"
            class="btn btn-ghost btn-sm rounded-xl"
            type="button"
            :disabled="isLoading"
            @click="refreshSamples(true)"
          >
            <span v-if="isLoading" class="loading loading-spinner loading-xs" />
            <Icon v-else name="kind-icon:refresh" class="h-4 w-4" />
            <span class="hidden sm:inline">Refresh</span>
          </button>
        </div>
      </div>

      <div
        v-if="showControls"
        class="grid grid-cols-1 gap-2 lg:grid-cols-[auto_auto_auto_minmax(0,1fr)_auto]"
      >
        <!-- Mature toggle is admin-only in the UI; regular users flip
             showMature from their account settings instead. -->
        <label
          v-if="userStore.isAdmin"
          class="label cursor-pointer justify-between rounded-2xl border border-base-300 bg-base-100 px-4 py-2"
        >
          <span class="label-text font-bold">Show Mature</span>

          <input
            v-model="showMature"
            type="checkbox"
            class="toggle toggle-accent toggle-sm"
          />
        </label>

        <select
          v-model="selectedScope"
          class="select select-bordered select-sm w-full bg-base-100 lg:w-auto"
          aria-label="Filter samples by scope"
        >
          <option value="all">All entries</option>
          <option value="mine">Mine</option>
          <option value="public">Public</option>
        </select>

        <select
          v-model="selectedType"
          class="select select-bordered select-sm w-full bg-base-100 lg:w-auto"
          aria-label="Filter samples by type"
        >
          <option value="all">All types</option>

          <option v-for="type in sampleTypes" :key="type" :value="type">
            {{ type }}
          </option>
        </select>

        <input
          v-model="searchQuery"
          type="search"
          aria-label="Search samples"
          placeholder="Search samples..."
          class="input input-bordered input-sm w-full bg-base-100"
        />

        <button
          class="btn btn-ghost btn-sm rounded-xl lg:w-auto"
          type="button"
          :disabled="!sampleStore.selectedSample"
          @click="clearSelectedSample"
        >
          <Icon name="kind-icon:x" class="h-4 w-4" />
          Clear
        </button>
      </div>
    </header>

    <!-- Inline add/edit panel; the same add-sample component also lives on
         the manager's add tab. -->
    <section
      v-if="showSampleForm"
      class="shrink-0 rounded-2xl border border-primary/30 bg-base-100 p-3 shadow-md"
    >
      <div class="mb-3 flex items-center justify-between gap-3">
        <h3 class="truncate text-base font-black text-primary">
          {{ formTitle }}
        </h3>

        <button
          class="btn btn-ghost btn-sm rounded-xl"
          type="button"
          @click="closeSampleForm"
        >
          <Icon name="kind-icon:x" class="h-4 w-4" />
          <span class="hidden sm:inline">Close</span>
        </button>
      </div>

      <add-sample
        :mode="formMode"
        @saved="handleSampleSaved"
        @cancel="closeSampleForm"
      />
    </section>

    <section class="min-h-0 flex-1 overflow-auto">
      <div
        v-if="isLoading"
        class="flex h-full items-center justify-center py-12"
      >
        <span class="loading loading-spinner loading-lg text-primary" />
      </div>

      <div
        v-else-if="sampleStore.error"
        class="flex h-full items-center justify-center rounded-2xl border border-error/40 bg-error/10 p-6 text-center text-error"
      >
        <p class="text-lg font-bold">
          {{ sampleStore.error }}
        </p>
      </div>

      <div
        v-else-if="filteredSamples.length === 0"
        class="flex h-full flex-col items-center justify-center gap-3 rounded-2xl border border-base-300 bg-base-200 p-6 text-center text-base-content/60"
      >
        <Icon name="kind-icon:box" class="h-10 w-10" />

        <p class="text-lg font-bold">No samples found.</p>

        <p class="max-w-xl text-sm opacity-70">
          No public or owned samples match this gallery.
        </p>

        <button
          v-if="allowAdd"
          class="btn btn-primary btn-sm rounded-xl"
          type="button"
          @click="startAddingSample"
        >
          <Icon name="kind-icon:plus" class="h-4 w-4" />
          Add Sample
        </button>
      </div>

      <div v-else :class="layoutClass">
        <sample-card
          v-for="sample in filteredSamples"
          :key="sample.id"
          :sample="sample"
          :selected="sampleStore.selectedSample?.id === sample.id"
          :compact="isCompact"
          :show-actions="showCardActions"
          :show-description="showDescriptions"
          :show-meta="showMeta"
          :allow-edit="allowEdit"
          :allow-delete="allowDelete"
          @select="selectSample"
          @edit="startEditingSampleById"
          @delete="handleSampleDeleted"
        />
      </div>
    </section>
  </section>
</template>

<script setup lang="ts">
import { computed, onMounted, ref } from 'vue'
import { useSampleStore, type Sample } from '@/stores/sampleStore'
import { useUserStore } from '@/stores/userStore'

// Need a dropdown variant too? Copy the isDropdownMode branch from
// /components/rewards/reward-gallery.vue.
type GalleryVariant = 'grid' | 'row'

const props = withDefaults(
  defineProps<{
    variant?: GalleryVariant
    title?: string
    subtitle?: string
    showHeader?: boolean
    showControls?: boolean
    showCardActions?: boolean
    showDescriptions?: boolean
    showMeta?: boolean
    allowAdd?: boolean
    allowEdit?: boolean
    allowDelete?: boolean
    allowRefresh?: boolean
    compact?: boolean
    autoLoad?: boolean
  }>(),
  {
    variant: 'grid',
    title: 'Samples',
    subtitle: 'Pick a sample entry.',
    showHeader: true,
    showControls: true,
    showCardActions: true,
    showDescriptions: true,
    showMeta: true,
    allowAdd: true,
    allowEdit: true,
    allowDelete: true,
    allowRefresh: true,
    compact: false,
    autoLoad: true,
  },
)

const sampleStore = useSampleStore()
const userStore = useUserStore()

const selectedScope = ref<'all' | 'mine' | 'public'>('all')
const selectedType = ref('all')
const searchQuery = ref('')
const isLoading = ref(false)
const showSampleForm = ref(false)
const formMode = ref<'add' | 'edit'>('add')

const isCompact = computed(() => {
  return props.compact || props.variant === 'row'
})

const layoutClass = computed(() => {
  return props.variant === 'row' ? 'sample-row' : 'sample-grid'
})

const formTitle = computed(() => {
  return formMode.value === 'edit' ? 'Edit Sample' : 'Add Sample'
})

const selectedSampleTitle = computed(() => {
  return sampleStore.selectedSample?.title || 'Untitled Sample'
})

const currentUserId = computed(() => {
  return userStore.userId ?? userStore.user?.id ?? null
})

// Writing the toggle persists the preference on the user record.
const showMature = computed({
  get: () => userStore.user?.showMature ?? userStore.showMature ?? false,
  set: async (value: boolean) => {
    if (!userStore.user) return

    await userStore.updateUser({ showMature: value })
  },
})

// Visibility gates come FIRST and apply to every variant: non-admins only
// see public rows or their own, and isMature rows are hidden unless the
// user opted in (admins bypass both). Required for every isMature model.
// The "snapshot data" indicator is global (navigation/snapshot-mode-banner
// reads snapshotLoader), so the gallery itself never renders one.
const gallerySamples = computed<Sample[]>(() => {
  let samples = sampleStore.samples

  if (!userStore.isAdmin) {
    samples = samples.filter((sample) => {
      return sample.isPublic || sample.userId === currentUserId.value
    })
  }

  if (!showMature.value) {
    samples = samples.filter((sample) => !sample.isMature)
  }

  return samples
})

const sampleTypes = computed(() => {
  const set = new Set<string>()

  for (const sample of gallerySamples.value) {
    const type = sample.type?.trim()

    if (type) {
      set.add(type)
    }
  }

  return Array.from(set).sort()
})

// User-facing filters (scope/type/search) layer on top of the gates.
const filteredSamples = computed<Sample[]>(() => {
  let samples = gallerySamples.value

  if (selectedScope.value === 'mine') {
    samples = samples.filter((sample) => {
      return sample.userId === currentUserId.value
    })
  }

  if (selectedScope.value === 'public') {
    samples = samples.filter((sample) => sample.isPublic)
  }

  if (selectedType.value !== 'all') {
    samples = samples.filter((sample) => sample.type === selectedType.value)
  }

  const query = searchQuery.value.trim().toLowerCase()

  if (query) {
    samples = samples.filter((sample) => {
      return (
        (sample.title || '').toLowerCase().includes(query) ||
        (sample.label || '').toLowerCase().includes(query) ||
        (sample.type || '').toLowerCase().includes(query) ||
        (sample.designer || '').toLowerCase().includes(query) ||
        (sample.description || '').toLowerCase().includes(query)
      )
    })
  }

  return samples
})

onMounted(async () => {
  if (props.autoLoad) {
    await refreshSamples()
  }
})

async function refreshSamples(force = false) {
  isLoading.value = true

  try {
    await sampleStore.initialize({
      force,
      fetchRemote: true,
    })
  } finally {
    isLoading.value = false
  }
}

async function selectSample(id: number) {
  await sampleStore.selectSample(id)
}

function startAddingSample() {
  sampleStore.startAddingSample()
  formMode.value = 'add'
  showSampleForm.value = true
}

async function startEditingSampleById(id: number) {
  const sample = await sampleStore.startEditingSample(id)

  if (!sample) return

  formMode.value = 'edit'
  showSampleForm.value = true
}

function clearSelectedSample() {
  sampleStore.deselectSample()
  showSampleForm.value = false
}

function closeSampleForm() {
  showSampleForm.value = false
}

async function handleSampleSaved() {
  showSampleForm.value = false
  await refreshSamples(true)
}

function handleSampleDeleted(id: number) {
  if (sampleStore.selectedSample?.id === id) {
    sampleStore.deselectSample()
  }
}
</script>
