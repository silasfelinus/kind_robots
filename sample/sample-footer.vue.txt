<!-- /components/navigation/sample-footer.vue -->
<template>
  <div
    v-if="footerState !== 'hidden'"
    class="flex h-full w-full min-h-0 overflow-hidden rounded-2xl border border-base-300 bg-base-200/80 p-2 shadow-inner md:p-3"
  >
    <div v-if="isCompact" class="flex h-full w-full min-h-0">
      <div
        class="grid h-full w-full min-h-0 grid-cols-[auto_1fr_auto] items-stretch gap-2 rounded-2xl border border-base-300 bg-base-100 p-2"
      >
        <div class="flex items-center gap-2 px-2">
          <Icon name="kind-icon:box" class="h-6 w-6 text-primary" />
          <span class="hidden text-sm font-bold sm:inline">
            Samples
          </span>
        </div>

        <select
          v-model.number="selectedId"
          class="select select-bordered select-sm h-full min-h-0 w-full rounded-2xl"
          @change="selectSample"
        >
          <option :value="0">Choose sample</option>
          <option
            v-for="sample in sampleStore.items"
            :key="sample.id"
            :value="sample.id"
          >
            {{ sample.title || `Sample ${sample.id}` }}
          </option>
        </select>

        <button
          class="btn h-full min-h-0 w-24 bg-primary font-semibold text-white hover:bg-primary/90"
          @click="openSampleMode"
        >
          Open
        </button>
      </div>
    </div>

    <div
      v-else
      class="grid h-full w-full min-h-0 grid-cols-1 gap-3 overflow-hidden xl:grid-cols-[18rem_minmax(0,1fr)]"
    >
      <section
        class="flex min-h-0 flex-col overflow-hidden rounded-2xl border border-base-300 bg-base-100 p-3 shadow"
      >
        <div class="flex items-center justify-between gap-2">
          <h2 class="text-base font-semibold">📦 Sample Footer</h2>

          <button
            class="btn btn-sm btn-primary font-semibold text-white"
            @click="openSampleMode"
          >
            Open Samples
          </button>
        </div>

        <div class="mt-3 min-h-0 flex-1 overflow-y-auto pr-1">
          <button
            v-for="sample in sampleStore.items"
            :key="sample.id"
            type="button"
            class="mb-2 flex w-full items-start gap-3 rounded-2xl border p-3 text-left transition"
            :class="
              sampleStore.selected?.id === sample.id
                ? 'border-primary bg-primary/10'
                : 'border-base-300 bg-base-200/60 hover:bg-base-200'
            "
            @click="selectSampleById(sample.id)"
          >
            <Icon
              :name="sample.icon || 'kind-icon:box'"
              class="mt-0.5 h-7 w-7 shrink-0 text-primary"
            />

            <div class="min-w-0 flex-1">
              <div class="truncate text-sm font-bold">
                {{ sample.title || `Sample ${sample.id}` }}
              </div>

              <div class="truncate text-xs text-base-content/55">
                {{ sample.type || 'No type' }}
              </div>
            </div>
          </button>
        </div>
      </section>

      <section
        class="flex min-h-0 flex-col overflow-hidden rounded-2xl border border-base-300 bg-base-100 p-3 shadow"
      >
        <div
          class="mb-2 text-sm font-semibold uppercase tracking-wide text-base-content/70"
        >
          Selected Sample
        </div>

        <div
          class="min-h-0 flex-1 overflow-y-auto rounded-2xl border border-base-300 bg-base-200 p-3 text-sm leading-relaxed"
        >
          <div v-if="sampleStore.selected" class="space-y-2">
            <div class="flex items-center gap-2">
              <Icon
                :name="sampleStore.selected.icon || 'kind-icon:box'"
                class="h-8 w-8 text-primary"
              />

              <div>
                <div class="font-bold">
                  {{ sampleStore.selected.title || 'Untitled Sample' }}
                </div>

                <div class="text-xs text-base-content/60">
                  {{ sampleStore.selected.type || 'No type' }}
                </div>
              </div>
            </div>

            <p class="text-base-content/70">
              {{ sampleStore.selected.description || sampleStore.selected.label || 'No description yet.' }}
            </p>
          </div>

          <div v-else>
            No sample selected yet.
          </div>
        </div>
      </section>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed, onMounted, ref, watch } from 'vue'
import { useRouter } from 'vue-router'
import { useDisplayStore } from '@/stores/displayStore'
import { useNavStore } from '@/stores/navStore'
import { useSampleModelStore } from '@/stores/sampleModelStore'

const router = useRouter()
const displayStore = useDisplayStore()
const navStore = useNavStore()
const sampleStore = useSampleModelStore()

const dashboardKey = 'sample' as const

const footerState = computed(() => displayStore.footerState)
const isCompact = computed(() => footerState.value === 'compact')

const selectedId = ref(0)

function selectSample() {
  if (!selectedId.value) {
    sampleStore.deselectModel()
    return
  }

  sampleStore.selectModel(selectedId.value)
}

function selectSampleById(id: number) {
  selectedId.value = id
  sampleStore.selectModel(id)
}

async function openSampleMode() {
  navStore.setDashboardTab(dashboardKey, 'overview')
  await router.push('/samples')
}

watch(
  () => sampleStore.selected?.id,
  (id) => {
    selectedId.value = id ?? 0
  },
)

onMounted(async () => {
  await sampleStore.initialize({
    fetchRemote: true,
  })

  selectedId.value = sampleStore.selected?.id ?? 0
})
</script>
