<!-- /components/sample/sample-manager.vue -->
<!-- TEMPLATE — the channel manager: a tab switcher driven by the dashboard
     registry. Modeled on /components/bots/bot-manager.vue. It composes the
     gallery/add/interact children; it does not re-implement them. -->
<template>
  <section class="flex h-full min-h-0 w-full flex-col overflow-hidden">
    <div
      v-if="isLoadingManager || managerError"
      class="mb-3 flex shrink-0 flex-wrap items-center justify-between gap-2 rounded-2xl border border-base-300 bg-base-100 p-3 text-sm shadow"
    >
      <p
        class="min-w-0 flex-1 text-base-content/70"
        :class="managerError ? 'text-error' : ''"
      >
        {{ managerError || 'Loading samples...' }}
      </p>

      <button
        type="button"
        class="btn btn-sm rounded-2xl"
        :class="managerError ? 'btn-error' : 'btn-ghost'"
        :disabled="isLoadingManager"
        @click="refreshManagerData"
      >
        <span
          v-if="isLoadingManager"
          class="loading loading-spinner loading-xs"
        />
        <Icon v-else name="kind-icon:refresh" class="h-4 w-4" />
        Refresh
      </button>
    </div>

    <sample-gallery
      v-if="activeTab === 'gallery'"
      class="min-h-0 flex-1"
      title="Sample Gallery"
      subtitle="Select, add, edit, or delete sample entries."
    />

    <section
      v-else-if="activeTab === 'add'"
      class="min-h-0 flex-1 overflow-y-auto rounded-2xl border border-base-300 bg-base-100 p-4"
    >
      <div class="mb-4 flex items-start justify-between gap-3">
        <div class="min-w-0">
          <h2 class="text-xl font-black text-primary">Add Sample</h2>

          <p class="mt-1 text-sm text-base-content/60">
            Create a new sample entry.
          </p>
        </div>

        <button
          class="btn btn-ghost btn-sm rounded-xl"
          type="button"
          @click="goToGallery"
        >
          <Icon name="kind-icon:arrow-left" class="h-4 w-4" />
          <span class="hidden sm:inline">Gallery</span>
        </button>
      </div>

      <add-sample
        :mode="sampleFormMode"
        @saved="handleSampleSaved"
        @cancel="goToGallery"
      />
    </section>

    <sample-interact
      v-else-if="activeTab === 'interact'"
      class="min-h-0 flex-1"
    />

    <div
      v-else
      class="flex min-h-0 flex-1 flex-col items-center justify-center gap-3 rounded-2xl border border-warning/40 bg-warning/10 p-4 text-center text-warning"
    >
      <Icon name="kind-icon:warning" class="h-10 w-10" />

      <div>
        <p class="text-lg font-black">Unknown sample tab: {{ activeTab }}</p>

        <p class="mt-1 text-sm opacity-80">
          Expected one of: {{ validTabLabels }}
        </p>
      </div>

      <button
        class="btn btn-warning btn-sm rounded-xl"
        type="button"
        @click="goToDefaultTab"
      >
        Go to {{ defaultTabLabel }}
      </button>
    </div>
  </section>
</template>

<script setup lang="ts">
import { computed, onMounted, ref } from 'vue'
import { useSampleStore } from '@/stores/sampleStore'
import { useNavStore } from '@/stores/navStore'
import {
  getDashboardConfig,
  getDashboardDefaultTab,
  getDashboardTabs,
  isDashboardTabKey,
  type DashboardKey,
} from '@/stores/helpers/dashboardHelper'

// 'sample' must be a registered DashboardKey — add the channel entry and its
// tabs (gallery, add, interact) in /stores/helpers/dashboardHelper.ts before
// this compiles. See sample/README.md, "Adding a new channel".
const dashboardKey: DashboardKey = 'sample'

const sampleStore = useSampleStore()
const navStore = useNavStore()

const isLoadingManager = ref(false)
const managerError = ref<string | null>(null)

const dashboardConfig = computed(() => {
  return getDashboardConfig(dashboardKey)
})

const dashboardTabs = computed(() => {
  return getDashboardTabs(dashboardKey)
})

const defaultTab = computed(() => {
  return getDashboardDefaultTab(dashboardKey)
})

// The nav store owns which tab is active; the manager just renders it and
// falls back to the registry default for unknown values.
const activeTab = computed(() => {
  const selectedTab = navStore.getDashboardTab(dashboardKey)

  if (isDashboardTabKey(dashboardKey, selectedTab)) {
    return selectedTab
  }

  return defaultTab.value
})

// The add tab doubles as the edit form when a row is selected.
const sampleFormMode = computed<'add' | 'edit'>(() => {
  return sampleStore.selectedSample ? 'edit' : 'add'
})

const validTabLabels = computed(() => {
  return dashboardTabs.value.map((tab) => tab.key).join(', ')
})

const defaultTabLabel = computed(() => {
  return (
    dashboardTabs.value.find((tab) => tab.key === defaultTab.value)?.label ||
    dashboardConfig.value.label
  )
})

async function loadManagerData(force = false) {
  isLoadingManager.value = true
  managerError.value = null

  try {
    await sampleStore.initialize({ force, fetchRemote: true })
  } catch (error) {
    managerError.value =
      error instanceof Error ? error.message : 'Failed to load samples.'
  } finally {
    isLoadingManager.value = false
  }
}

async function refreshManagerData() {
  await loadManagerData(true)
}

function goToGallery() {
  navStore.setDashboardTab(dashboardKey, 'gallery')
}

function goToDefaultTab() {
  navStore.setDashboardTab(dashboardKey, defaultTab.value)
}

async function handleSampleSaved() {
  await loadManagerData(true)
  goToGallery()
}

onMounted(async () => {
  await loadManagerData()
})
</script>
