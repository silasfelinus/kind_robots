<!-- /components/content/sample/sample-card.vue -->
<template>
  <reactable-card
    :selected="activeSelected"
    :compact="compact"
    :show-reaction="showReaction"
    :target-id="sample.id"
    target-type="resource"
    reaction-category="RESOURCE"
    :target-title="sampleTitle"
    @select="selectSample"
  >
    <template #actions>
      <button
        v-if="showActions && allowEdit && (activeSelected || compact)"
        class="rounded-full bg-base-100 p-2 text-primary shadow transition hover:bg-primary hover:text-primary-content"
        type="button"
        title="Edit Sample"
        @click.stop="emit('edit', sample.id)"
      >
        <Icon name="kind-icon:pencil" class="h-4 w-4" />
      </button>

      <button
        v-if="showActions && allowDelete && (activeSelected || compact)"
        class="rounded-full bg-base-100 p-2 text-error shadow transition hover:bg-error hover:text-error-content"
        type="button"
        title="Delete Sample"
        @click.stop="deleteSample"
      >
        <Icon name="kind-icon:trash" class="h-4 w-4" />
      </button>
    </template>

    <div
      v-if="showImage"
      :class="[
        'relative flex items-center justify-center overflow-hidden rounded-2xl border border-base-300 bg-base-300',
        compact ? 'h-24 w-full' : 'h-36 w-full',
      ]"
    >
      <Icon
        :name="sample.icon || fallbackIcon"
        :class="compact ? 'h-12 w-12' : 'h-20 w-20'"
        class="text-primary/70"
      />

      <div class="absolute left-2 top-2 flex flex-wrap gap-1">
        <span
          v-if="activeSelected"
          class="badge badge-primary badge-sm"
        >
          Selected
        </span>

        <span
          class="badge badge-sm"
          :class="sample.isPublic ? 'badge-success' : 'badge-warning'"
        >
          {{ sample.isPublic ? 'Public' : 'Private' }}
        </span>
      </div>

      <div
        v-if="activeSelected"
        class="absolute bottom-2 right-2 rounded-full bg-primary p-2 text-primary-content shadow"
      >
        <Icon name="kind-icon:check" class="h-4 w-4" />
      </div>
    </div>

    <div class="flex min-w-0 flex-1 flex-col gap-2">
      <h2
        :class="[
          'font-black leading-tight text-base-content',
          compact ? 'line-clamp-1 text-base' : 'text-lg',
        ]"
        :title="sampleTitle"
      >
        {{ sampleTitle }}
      </h2>

      <p
        v-if="showDescription"
        :class="[
          'text-base-content/70',
          compact ? 'line-clamp-2 text-sm' : 'text-sm',
        ]"
      >
        {{ sample.description || sample.label || 'No description yet.' }}
      </p>

      <div v-if="showMeta" class="flex flex-wrap gap-2">
        <span class="badge badge-outline badge-sm">
          {{ sample.type || 'sample' }}
        </span>

        <span v-if="sample.label" class="badge badge-ghost badge-sm">
          {{ sample.label }}
        </span>

        <span v-if="sample.component" class="badge badge-primary badge-sm">
          {{ sample.component }}
        </span>

        <span v-if="sample.link" class="badge badge-secondary badge-sm">
          Link
        </span>
      </div>

      <button
        v-if="showSelectButton"
        class="btn btn-sm mt-auto rounded-xl"
        :class="activeSelected ? 'btn-primary text-white' : 'btn-outline'"
        type="button"
        @click.stop="selectSample"
      >
        <Icon name="kind-icon:check" class="h-4 w-4" />
        {{ activeSelected ? 'Selected' : 'Select' }}
      </button>

      <details
        v-if="showDebug"
        class="rounded-2xl border border-base-300 bg-base-100 p-2"
        @click.stop
      >
        <summary class="cursor-pointer text-xs font-bold text-base-content/70">
          Debug
        </summary>

        <pre class="mt-2 max-h-48 overflow-auto text-xs text-base-content/70">{{ JSON.stringify(sample, null, 2) }}</pre>
      </details>
    </div>
  </reactable-card>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import { useSampleModelStore } from '@/stores/sampleModelStore'

type SampleModel = {
  id: number
  title?: string | null
  type?: string | null
  label?: string | null
  icon?: string | null
  description?: string | null
  link?: string | null
  component?: string | null
  isPublic?: boolean | null
}

const props = withDefaults(
  defineProps<{
    sample: SampleModel
    selected?: boolean
    showImage?: boolean
    compact?: boolean
    showActions?: boolean
    showDescription?: boolean
    showMeta?: boolean
    showSelectButton?: boolean
    showReaction?: boolean
    showDebug?: boolean
    allowEdit?: boolean
    allowDelete?: boolean
    fallbackIcon?: string
  }>(),
  {
    selected: false,
    showImage: true,
    compact: false,
    showActions: true,
    showDescription: true,
    showMeta: true,
    showSelectButton: false,
    showReaction: true,
    showDebug: false,
    allowEdit: true,
    allowDelete: true,
    fallbackIcon: 'kind-icon:box',
  },
)

const emit = defineEmits<{
  edit: [id: number]
  delete: [id: number]
}>()

const sampleStore = useSampleModelStore()

const activeSelected = computed(() => {
  return props.selected || sampleStore.selectedModel?.id === props.sample.id
})

const sampleTitle = computed(() => {
  return props.sample.title || props.sample.label || `Sample #${props.sample.id}`
})

function selectSample() {
  sampleStore.selectModel(props.sample.id)
}

async function deleteSample() {
  const result = await sampleStore.deleteModel(props.sample.id)

  if (result?.success !== false) {
    emit('delete', props.sample.id)
  }
}
</script>
